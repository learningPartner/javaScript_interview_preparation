find Missing Number from array 
 
function findMissingNumber(arr) {
    const n = arr.length + 1;
    let missingElements = []; 
    for (let i = 1; i <= n; i++) {
        if (!arr.includes(i)) {
            missingElements.push(i)
        };
    }
    return missingElements;
}
console.log(findMissingNumber([1, 2, 4, 5, 7]))
